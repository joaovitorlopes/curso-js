# installing package:
- npx eslint --init
- npm i nodemon --save-dev
- npm i sucrase --save-dev
- npm i express
- npm i dotenv
- npm i sequelize

- npm i sequelize-cli --save-dev
- npm i bcryptjs
- npm i jsonwebtoken

# sequelize commands:
- "npx sequelize migration:create --name=NAME" to create an migration
- "npx sequelize db:migrate" to send migration
